  <script>

    function calculateBaselineCost(cryptoassets) {

      if (cryptoassets === 1) return 14000;

      if (cryptoassets <= 5) return 16000;

      if (cryptoassets <= 10) return 18000;

      if (cryptoassets <= 20) return 20000;

      if (cryptoassets <= 30) return 22000;

      if (cryptoassets <= 50) return 24000;

      if (cryptoassets <= 100) return 25000;

      return 27000;

    }


    function calculateRiskFactor(riskTokens) {

      if (riskTokens <= 0) return 1.0;

      if (riskTokens <= 5) return 1.2;

      if (riskTokens <= 10) return 1.4;

      if (riskTokens <= 20) return 1.6;

      if (riskTokens <= 40) return 1.8;

      if (riskTokens <= 80) return 2.0;

      return 2.3;

    }


    function calculateWebAppCost(webPages, emails, baselineCost) {

      let webAppFactor = 0;

      if (webPages >= 1 && webPages <= 5) webAppFactor += 0.2;

      else if (webPages <= 10) webAppFactor += 0.4;

      else if (webPages <= 20) webAppFactor += 0.6;

      else if (webPages <= 40) webAppFactor += 0.8;

      else if (webPages <= 80) webAppFactor += 1.0;

      else if (webPages > 80) webAppFactor += 1.5;


      let emailFactor = 0;

      if (emails >= 1 && emails <= 5) emailFactor += 0.05;

      else if (emails <= 10) emailFactor += 0.1;

      else if (emails <= 15) emailFactor += 0.15;

      else if (emails <= 20) emailFactor += 0.2;

      else if (emails <= 30) emailFactor += 0.3;

      else if (emails <= 40) emailFactor += 0.4;

      else if (emails <= 60) emailFactor += 0.6;


      return baselineCost * (webAppFactor + emailFactor);

    }


    function calculateSocialCost(socialChannelsFactor, socialPosts, baselineCost) {

      let postFactor = 0;

      if (socialPosts >= 6 && socialPosts <= 10) postFactor += 0.2;

      else if (socialPosts <= 15) postFactor += 0.4;

      else if (socialPosts <= 20) postFactor += 0.8;

      else if (socialPosts <= 30) postFactor += 1.2;

      else if (socialPosts <= 40) postFactor += 1.8;

      else if (socialPosts <= 60) postFactor += 2.5;


      return baselineCost * (socialChannelsFactor + postFactor);

    }


    function calculatePrice() {

      const cryptoassets = parseInt(document.getElementById('cryptoassets').value) || 0;

      const riskTokens = parseInt(document.getElementById('risk-tokens').value) || 0;

      const webPages = parseInt(document.getElementById('web-pages').value) || 0;

      const emails = parseInt(document.getElementById('emails').value) || 0;

      const socialChannelsFactor = parseFloat(document.getElementById('social-channels').value) || 0;

      const socialPosts = parseInt(document.getElementById('social-posts').value) || 0;


      const baselineCost = calculateBaselineCost(cryptoassets);

      const riskFactor = calculateRiskFactor(riskTokens);


      const webAppCost = calculateWebAppCost(webPages, emails, baselineCost);

      const socialCost = calculateSocialCost(socialChannelsFactor, socialPosts, baselineCost);


      const totalCost = baselineCost * riskFactor + webAppCost + socialCost;


      document.getElementById('result').innerText = `Total Price: £${totalCost.toFixed(2)}`;

    }
